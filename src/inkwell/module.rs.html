67/// This enum defines how to link a global variable or function in a module. The variant documentation is
68/// mostly taken straight from LLVM's own documentation except for some minor clarification.
69///
70/// It is illegal for a function declaration to have any linkage type other than external or extern_weak.
71///
72/// All Global Variables, Functions and Aliases can have one of the following DLL storage class: `DLLImport`
73/// & `DLLExport`.
74// REVIEW: Maybe this should go into it's own module?
75pub enum Linkage {
76    /// `Appending` linkage may only be applied to global variables of pointer to array type. When two global
77    /// variables with appending linkage are linked together, the two global arrays are appended together.
78    /// This is the LLVM, typesafe, equivalent of having the system linker append together "sections" with
79    /// identical names when .o files are linked. Unfortunately this doesn't correspond to any feature in .o
80    /// files, so it can only be used for variables like llvm.global_ctors which llvm interprets specially.
81    #[llvm_variant(LLVMAppendingLinkage)]
82    Appending,
83    /// Globals with `AvailableExternally` linkage are never emitted into the object file corresponding to
84    /// the LLVM module. From the linker's perspective, an `AvailableExternally` global is equivalent to an
85    /// external declaration. They exist to allow inlining and other optimizations to take place given
86    /// knowledge of the definition of the global, which is known to be somewhere outside the module. Globals
87    /// with `AvailableExternally` linkage are allowed to be discarded at will, and allow inlining and other
88    /// optimizations. This linkage type is only allowed on definitions, not declarations.
89    #[llvm_variant(LLVMAvailableExternallyLinkage)]
90    AvailableExternally,
91    /// `Common` linkage is most similar to "weak" linkage, but they are used for tentative definitions
92    /// in C, such as "int X;" at global scope. Symbols with Common linkage are merged in the same way as
93    /// weak symbols, and they may not be deleted if unreferenced. `Common` symbols may not have an explicit
94    /// section, must have a zero initializer, and may not be marked 'constant'. Functions and aliases may
95    /// not have `Common` linkage.
96    #[llvm_variant(LLVMCommonLinkage)]
97    Common,
98    /// `DLLExport` causes the compiler to provide a global pointer to a pointer in a DLL, so that it can be
99    /// referenced with the dllimport attribute. On Microsoft Windows targets, the pointer name is formed by
100    /// combining __imp_ and the function or variable name. Since this storage class exists for defining a dll
101    /// interface, the compiler, assembler and linker know it is externally referenced and must refrain from
102    /// deleting the symbol.
103    #[llvm_variant(LLVMDLLExportLinkage)]
104    DLLExport,
105    /// `DLLImport` causes the compiler to reference a function or variable via a global pointer to a pointer
106    /// that is set up by the DLL exporting the symbol. On Microsoft Windows targets, the pointer name is
107    /// formed by combining __imp_ and the function or variable name.
108    #[llvm_variant(LLVMDLLImportLinkage)]
109    DLLImport,
110    /// If none of the other identifiers are used, the global is externally visible, meaning that it
111    /// participates in linkage and can be used to resolve external symbol references.
112    #[llvm_variant(LLVMExternalLinkage)]
113    External,
114    /// The semantics of this linkage follow the ELF object file model: the symbol is weak until linked,
115    /// if not linked, the symbol becomes null instead of being an undefined reference.
116    #[llvm_variant(LLVMExternalWeakLinkage)]
117    ExternalWeak,
118    /// FIXME: Unknown linkage type
119    #[llvm_variant(LLVMGhostLinkage)]
120    Ghost,
121    /// Similar to private, but the value shows as a local symbol (STB_LOCAL in the case of ELF) in the object
122    /// file. This corresponds to the notion of the 'static' keyword in C.
123    #[llvm_variant(LLVMInternalLinkage)]
124    Internal,
125    /// FIXME: Unknown linkage type
126    #[llvm_variant(LLVMLinkerPrivateLinkage)]
127    LinkerPrivate,
128    /// FIXME: Unknown linkage type
129    #[llvm_variant(LLVMLinkerPrivateWeakLinkage)]
130    LinkerPrivateWeak,
131    /// Globals with `LinkOnceAny` linkage are merged with other globals of the same name when linkage occurs.
132    /// This can be used to implement some forms of inline functions, templates, or other code which must be
133    /// generated in each translation unit that uses it, but where the body may be overridden with a more
134    /// definitive definition later. Unreferenced `LinkOnceAny` globals are allowed to be discarded. Note that
135    /// `LinkOnceAny` linkage does not actually allow the optimizer to inline the body of this function into
136    /// callers because it doesn’t know if this definition of the function is the definitive definition within
137    /// the program or whether it will be overridden by a stronger definition. To enable inlining and other
138    /// optimizations, use `LinkOnceODR` linkage.
139    #[llvm_variant(LLVMLinkOnceAnyLinkage)]
140    LinkOnceAny,
141    /// FIXME: Unknown linkage type
142    #[llvm_variant(LLVMLinkOnceODRAutoHideLinkage)]
143    LinkOnceODRAutoHide,
144    /// Some languages allow differing globals to be merged, such as two functions with different semantics.
145    /// Other languages, such as C++, ensure that only equivalent globals are ever merged (the "one definition
146    /// rule" — "ODR"). Such languages can use the `LinkOnceODR` and `WeakODR` linkage types to indicate that
147    /// the global will only be merged with equivalent globals. These linkage types are otherwise the same
148    /// as their non-odr versions.
149    #[llvm_variant(LLVMLinkOnceODRLinkage)]
150    LinkOnceODR,
151    /// Global values with `Private` linkage are only directly accessible by objects in the current module.
152    /// In particular, linking code into a module with a private global value may cause the private to be
153    /// renamed as necessary to avoid collisions. Because the symbol is private to the module, all references
154    /// can be updated. This doesn’t show up in any symbol table in the object file.
155    #[llvm_variant(LLVMPrivateLinkage)]
156    Private,
157    /// `WeakAny` linkage has the same merging semantics as linkonce linkage, except that unreferenced globals
158    /// with weak linkage may not be discarded. This is used for globals that are declared WeakAny in C source code.
159    #[llvm_variant(LLVMWeakAnyLinkage)]
160    WeakAny,
161    /// Some languages allow differing globals to be merged, such as two functions with different semantics.
162    /// Other languages, such as C++, ensure that only equivalent globals are ever merged (the "one definition
163    /// rule" — "ODR"). Such languages can use the `LinkOnceODR` and `WeakODR` linkage types to indicate that
164    /// the global will only be merged with equivalent globals. These linkage types are otherwise the same
165    /// as their non-odr versions.
166    #[llvm_variant(LLVMWeakODRLinkage)]
167    WeakODR,
168}

202    /// Creates a function given its `name` and `ty`, adds it to the `Module`
203    /// and returns it.
204    ///
205    /// An optional `linkage` can be specified, without which the default value
206    /// `Linkage::ExternalLinkage` will be used.
207    ///
208    /// # Example
209    /// ```no_run
210    /// use inkwell::context::Context;
211    /// use inkwell::module::{Module, Linkage};
212    /// use inkwell::types::FunctionType;
213    ///
214    /// let context = Context::create();
215    /// let module = context.create_module("my_module");
216    ///
217    /// let fn_type = context.f32_type().fn_type(&[], false);
218    /// let fn_val = module.add_function("my_function", fn_type, None);
219    ///
220    /// assert_eq!(fn_val.get_name().to_str(), Ok("my_function"));
221    /// assert_eq!(fn_val.get_linkage(), Linkage::External);
222    /// ```
223    pub fn add_function(&self, name: &str, ty: FunctionType<'ctx>, linkage: Option<Linkage>) -> FunctionValue<'ctx> {
224        let c_string = to_c_str(name);
225        let fn_value = unsafe {
226            FunctionValue::new(LLVMAddFunction(self.module.get(), c_string.as_ptr(), ty.as_type_ref()))
227                .expect("add_function should always succeed in adding a new function")
228        };
229
230        if let Some(linkage) = linkage {
231            fn_value.set_linkage(linkage)
232        }
233
234        fn_value
235    }

448    /// Creates an `ExecutionEngine` from this `Module`.
449    ///
450    /// # Example
451    /// ```no_run
452    /// use inkwell::context::Context;
453    /// use inkwell::module::Module;
454    /// use inkwell::targets::{InitializationConfig, Target};
455    ///
456    /// Target::initialize_native(&InitializationConfig::default()).expect("Failed to initialize native target");
457    ///
458    /// let context = Context::create();
459    /// let module = context.create_module("my_module");
460    /// let execution_engine = module.create_execution_engine().unwrap();
461    ///
462    /// assert_eq!(module.get_context(), context);
463    /// ```
464    // SubType: ExecutionEngine<Basic?>
465    pub fn create_execution_engine(&self) -> Result<ExecutionEngine<'ctx>, LLVMString> {
466        Target::initialize_native(&InitializationConfig::default()).map_err(|mut err_string| {
467            err_string.push('\0');
468
469            LLVMString::create_from_str(&err_string)
470        })?;
471
472        if self.owned_by_ee.borrow().is_some() {
473            let string = "This module is already owned by an ExecutionEngine.\0";
474            return Err(LLVMString::create_from_str(string));
475        }
476
477        let mut execution_engine = MaybeUninit::uninit();
478        let mut err_string = MaybeUninit::uninit();
479        let code = unsafe {
480            // Takes ownership of module
481            LLVMCreateExecutionEngineForModule(
482                execution_engine.as_mut_ptr(),
483                self.module.get(),
484                err_string.as_mut_ptr(),
485            )
486        };
487
488        if code == 1 {
489            unsafe {
490                return Err(LLVMString::new(err_string.assume_init()));
491            }
492        }
493
494        let execution_engine = unsafe { execution_engine.assume_init() };
495        let execution_engine = unsafe { ExecutionEngine::new(Rc::new(execution_engine), false) };
496
497        *self.owned_by_ee.borrow_mut() = Some(execution_engine.clone());
498
499        Ok(execution_engine)
500    }

591        let mut err_string = MaybeUninit::uninit();

596                execution_engine.as_mut_ptr(),
597                self.module.get(),
598                opt_level as u32,
599                err_string.as_mut_ptr(),
600            )
601        };
602

604            unsafe {
605                return Err(LLVMString::new(err_string.assume_init()));
606            }
607        }
608

610        let execution_engine = unsafe { ExecutionEngine::new(Rc::new(execution_engine), true) };

617    /// Creates an MCJIT `ExecutionEngine` for this `Module` using a custom memory manager.
618    ///
619    /// # Parameters
620    ///
621    /// * `memory_manager` - Specifies how LLVM allocates and finalizes code and data sections.
622    ///   Implement the [`McjitMemoryManager`] trait to customize these operations.
623    /// * `opt_level` - Sets the desired optimization level (e.g. `None`, `Less`, `Default`, `Aggressive`).
624    ///   Higher levels generally produce faster code at the expense of longer compilation times.
625    /// * `code_model` - Determines how code addresses are represented. Common values include
626    ///   `CodeModel::Default` or `CodeModel::JITDefault`. This impacts the generated machine code layout.
627    /// * `no_frame_pointer_elim` - If true, frame pointer elimination is disabled. This may assist
628    ///   with certain debugging or profiling tasks but can incur a performance cost.
629    /// * `enable_fast_isel` - If true, uses a faster instruction selector where possible. This can
630    ///   improve compilation speed, though it may produce less optimized code in some cases.
631    ///
632    /// # Returns
633    ///
634    /// Returns a newly created [`ExecutionEngine`] for MCJIT on success. Returns an error if:
635    /// - The native target fails to initialize,
636    /// - The `Module` is already owned by another `ExecutionEngine`,
637    /// - Or MCJIT fails to create the engine (in which case an error string is returned from LLVM).
638    ///
639    /// # Notes
640    ///
641    /// Using a custom memory manager can help intercept or manage allocations for specific
642    /// sections (for example, capturing `.llvm_stackmaps` or applying custom permissions).
643    /// For details, refer to the [`McjitMemoryManager`] documentation.
644    ///
645    /// # Safety
646    ///
647    /// The returned [`ExecutionEngine`] takes ownership of the memory manager. Do not move
648    /// or free the `memory_manager` after calling this method. When the `ExecutionEngine`
649    /// is dropped, LLVM will destroy the memory manager by calling
650    /// [`McjitMemoryManager::destroy()`] and freeing its adapter.
651    pub fn create_mcjit_execution_engine_with_memory_manager(
652        &self,
653        memory_manager: impl McjitMemoryManager + 'static,
654        opt_level: OptimizationLevel,
655        code_model: CodeModel,
656        no_frame_pointer_elim: bool,
657        enable_fast_isel: bool,
658    ) -> Result<ExecutionEngine<'ctx>, LLVMString> {
659        use std::mem::MaybeUninit;
660        // ...
661
662        // 1) Initialize the native target
663        Target::initialize_native(&InitializationConfig::default()).map_err(|mut err_string| {
664            err_string.push('\0');
665            LLVMString::create_from_str(&err_string)
666        })?;
667
668        // Check if the module is already owned by an ExecutionEngine
669        if self.owned_by_ee.borrow().is_some() {
670            let string = "This module is already owned by an ExecutionEngine.\0";
671            return Err(LLVMString::create_from_str(string));
672        }
673
674        // 2) Box the memory_manager into a MemoryManagerAdapter
675        let adapter = MemoryManagerAdapter {
676            memory_manager: Box::new(memory_manager),
677        };
678        let adapter_box = Box::new(adapter);
679        // Convert the Box into a raw pointer for LLVM.
680        // In `destroy_adapter`, we use `Box::from_raw` to safely reclaim ownership.
681        let opaque = Box::into_raw(adapter_box) as *mut c_void;
682
683        // 3) Create the LLVMMCJITMemoryManager using the custom callbacks
684        let mmgr = unsafe {
685            LLVMCreateSimpleMCJITMemoryManager(
686                opaque,
687                allocate_code_section_adapter,
688                allocate_data_section_adapter,
689                finalize_memory_adapter,
690                Some(destroy_adapter),
691            )
692        };
693        if mmgr.is_null() {
694            let msg = "Failed to create SimpleMCJITMemoryManager.\0";
695            return Err(LLVMString::create_from_str(msg));
696        }
697
698        // 4) Build LLVMMCJITCompilerOptions
699        let mut options_uninit = MaybeUninit::<llvm_sys::execution_engine::LLVMMCJITCompilerOptions>::zeroed();
700        unsafe {
701            // Ensure defaults are initialized
702            llvm_sys::execution_engine::LLVMInitializeMCJITCompilerOptions(
703                options_uninit.as_mut_ptr(),
704                std::mem::size_of::<llvm_sys::execution_engine::LLVMMCJITCompilerOptions>(),
705            );
706        }
707        let mut options = unsafe { options_uninit.assume_init() };
708
709        // Override fields
710        options.OptLevel = opt_level as u32;
711        options.CodeModel = code_model.into();
712        options.NoFramePointerElim = no_frame_pointer_elim as i32;
713        options.EnableFastISel = enable_fast_isel as i32;
714        options.MCJMM = mmgr;
715
716        // 5) Create MCJIT
717        let mut execution_engine = MaybeUninit::uninit();
718        let mut err_string = MaybeUninit::uninit();
719        let code = unsafe {
720            llvm_sys::execution_engine::LLVMCreateMCJITCompilerForModule(
721                execution_engine.as_mut_ptr(),
722                self.module.get(),
723                &mut options,
724                std::mem::size_of::<llvm_sys::execution_engine::LLVMMCJITCompilerOptions>(),
725                err_string.as_mut_ptr(),
726            )
727        };
728
729        // If creation fails, extract the error string
730        if code == 1 {
731            unsafe {
732                return Err(LLVMString::new(err_string.assume_init()));
733            }
734        }
735
736        // Otherwise, it succeeded, so wrap the raw pointer
737        let execution_engine = unsafe { execution_engine.assume_init() };
738        let execution_engine = unsafe { ExecutionEngine::new(Rc::new(execution_engine), true) };
739
740        *self.owned_by_ee.borrow_mut() = Some(execution_engine.clone());
741
742        Ok(execution_engine)
743    }

1422    ///
1423    /// # Example
1424    ///
1425    /// ```no_run
1426    /// use inkwell::context::Context;
1427    ///
1428    /// let context = Context::create();

1430    /// let module2 = context.create_module("mod2");
1431    ///
1432    /// assert!(module.link_in_module(module2).is_ok());
1433    /// ```
1434    pub fn link_in_module(&self, other: Self) -> Result<(), LLVMString> {
1435        if other.owned_by_ee.borrow().is_some() {
1436            let string = "Cannot link a module which is already owned by an ExecutionEngine.\0";
1437            return Err(LLVMString::create_from_str(string));
1438        }
1439
1440        use crate::support::error_handling::get_error_str_diagnostic_handler;
1441        use libc::c_void;
1442        use llvm_sys::linker::LLVMLinkModules2;
1443
1444        let context = self.get_context();
1445
1446        let mut char_ptr: *mut ::libc::c_char = ptr::null_mut();
1447        let char_ptr_ptr = &mut char_ptr as *mut *mut ::libc::c_char as *mut *mut c_void as *mut c_void;
1448
1449        // Newer LLVM versions don't use an out ptr anymore which was really straightforward...
1450        // Here we assign an error handler to extract the error message, if any, for us.
1451        context.set_diagnostic_handler(get_error_str_diagnostic_handler, char_ptr_ptr);
1452
1453        let code = unsafe { LLVMLinkModules2(self.module.get(), other.module.get()) };
1454
1455        forget(other);
1456
1457        if code == 1 {
1458            debug_assert!(!char_ptr.is_null());
1459
1460            unsafe { Err(LLVMString::new(char_ptr)) }
1461        } else {
1462            Ok(())
1463        }
1464    }

1695#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
1696/// Defines the operational behavior for a module wide flag. This documentation comes directly
1697/// from the LLVM docs
1698pub enum FlagBehavior {
1699    /// Emits an error if two values disagree, otherwise the resulting value is that of the operands.
1700    #[llvm_variant(LLVMModuleFlagBehaviorError)]
1701    Error,
1702    /// Emits a warning if two values disagree. The result value will be the operand for the
1703    /// flag from the first module being linked.
1704    #[llvm_variant(LLVMModuleFlagBehaviorWarning)]
1705    Warning,
1706    /// Adds a requirement that another module flag be present and have a specified value after
1707    /// linking is performed. The value must be a metadata pair, where the first element of the
1708    /// pair is the ID of the module flag to be restricted, and the second element of the pair
1709    /// is the value the module flag should be restricted to. This behavior can be used to
1710    /// restrict the allowable results (via triggering of an error) of linking IDs with the
1711    /// **Override** behavior.
1712    #[llvm_variant(LLVMModuleFlagBehaviorRequire)]
1713    Require,
1714    /// Uses the specified value, regardless of the behavior or value of the other module. If
1715    /// both modules specify **Override**, but the values differ, an error will be emitted.
1716    #[llvm_variant(LLVMModuleFlagBehaviorOverride)]
1717    Override,
1718    /// Appends the two values, which are required to be metadata nodes.
1719    #[llvm_variant(LLVMModuleFlagBehaviorAppend)]
1720    Append,
1721    /// Appends the two values, which are required to be metadata nodes. However, duplicate
1722    /// entries in the second list are dropped during the append operation.
1723    #[llvm_variant(LLVMModuleFlagBehaviorAppendUnique)]
1724    AppendUnique,
1725}
