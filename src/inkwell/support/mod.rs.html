19/// An owned LLVM String. Also known as a LLVM Message
20#[derive(Eq)]
21pub struct LLVMString {
22    pub(crate) ptr: NonNull<c_char>,
23}
24const _: () = assert_niche::<LLVMString>();
25
26unsafe impl Send for LLVMString {}
27unsafe impl Sync for LLVMString {}
28
29impl LLVMString {
30    pub(crate) unsafe fn new(ptr: *const c_char) -> Self {
31        assert!(!ptr.is_null());
32        LLVMString {
33            ptr: unsafe { NonNull::new_unchecked(ptr.cast_mut()) },
34        }
35    }
36
37    /// This is a convenience method for creating a Rust `String`,
38    /// however; it *will* reallocate. `LLVMString` should be used
39    /// as much as possible to save memory since it is allocated by
40    /// LLVM. It's essentially a `CString` with a custom LLVM
41    /// deallocator
42    #[allow(clippy::inherent_to_string_shadow_display)]
43    pub fn to_string(&self) -> String {
44        (*self).to_string_lossy().into_owned()
45    }
46
47    /// This method will allocate a c string through LLVM
48    pub(crate) fn create_from_c_str(string: &CStr) -> LLVMString {
49        unsafe { LLVMString::new(LLVMCreateMessage(string.as_ptr() as *const _)) }
50    }
51
52    /// This method will allocate a c string through LLVM
53    pub(crate) fn create_from_str(string: &str) -> LLVMString {
54        debug_assert_eq!(string.as_bytes()[string.len() - 1], 0);
55
56        unsafe { LLVMString::new(LLVMCreateMessage(string.as_ptr() as *const _)) }
57    }
58}

105// Similar to Cow; however does not provide ability to clone
106// since memory is allocated by LLVM. Could use a better name
107// too. This is meant to be an internal wrapper only. Maybe
108// belongs in a private utils module.
109#[derive(Eq)]
110pub(crate) enum LLVMStringOrRaw {
111    Owned(LLVMString),
112    Borrowed(NonNull<c_char>),
113}
114
115impl LLVMStringOrRaw {
116    pub fn as_str(&self) -> &CStr {
117        match self {
118            LLVMStringOrRaw::Owned(llvm_string) => llvm_string.deref(),
119            LLVMStringOrRaw::Borrowed(ptr) => unsafe { CStr::from_ptr(ptr.as_ptr()) },
120        }
121    }
122}
123
124impl PartialEq for LLVMStringOrRaw {
125    fn eq(&self, other: &LLVMStringOrRaw) -> bool {
126        self.as_str() == other.as_str()
127    }
128}

140/// Returns the major, minor, and patch version of the LLVM in use
141#[llvm_versions(16..)]
142pub fn get_llvm_version() -> (u32, u32, u32) {
143    let mut major: u32 = 0;
144    let mut minor: u32 = 0;
145    let mut patch: u32 = 0;
146
147    unsafe { LLVMGetVersion(&mut major, &mut minor, &mut patch) };
148
149    (major, minor, patch)
150}

152/// Possible errors that can occur when loading a library
153#[derive(thiserror::Error, Debug, PartialEq, Eq, Clone, Copy)]
154pub enum LoadLibraryError {
155    /// The given path could not be converted to a [`&str`]
156    #[error("The given path could not be converted to a `&str`")]
157    UnicodeError,
158    /// The given path could not be loaded as a library
159    #[error("The given path could not be loaded as a library")]
160    LoadingError,
161}

163/// Permanently load the dynamic library at the given `path`.
164///
165/// It is safe to call this function multiple times for the same library.
166pub fn load_library_permanently(path: &Path) -> Result<(), LoadLibraryError> {
167    let filename = to_c_str(path.to_str().ok_or(LoadLibraryError::UnicodeError)?);
168
169    let error = unsafe { LLVMLoadLibraryPermanently(filename.as_ptr()) == 1 };
170    if error {
171        return Err(LoadLibraryError::LoadingError);
172    }
173
174    Ok(())
175}

178fn test_load_library_permanently() {
179    assert_eq!(
180        load_library_permanently(Path::new("missing.dll")),
181        Err(LoadLibraryError::LoadingError)

190/// Search through all previously loaded dynamic libraries for `symbol`.
191///
192/// Returns an address of the symbol, if found
193pub fn search_for_address_of_symbol(symbol: &str) -> Option<usize> {
194    let symbol = to_c_str(symbol);
195
196    let address = unsafe { LLVMSearchForAddressOfSymbol(symbol.as_ptr()) };
197    if address.is_null() {
198        return None;
199    }
200    Some(address as usize)
201}

206    load_visible_symbols();

222/// This function takes in a Rust string and either:
223///
224/// A) Finds a terminating null byte in the Rust string and can reference it directly like a C string.
225///
226/// B) Finds no null byte and allocates a new C string based on the input Rust string.
227#[inline]
228pub(crate) fn to_c_str(mut s: &str) -> Cow<'_, CStr> {
229    if s.is_empty() {
230        s = "\0";
231    }
232
233    match CStr::from_bytes_until_nul(s.as_bytes()) {
234        Ok(c) => Cow::from(c),
235        // SAFETY: No internal 0 byte since already `FromBytesUntilNulError`
236        Err(_) => unsafe { Cow::from(CString::new(s.as_bytes()).unwrap_unchecked()) },
237    }
238}

240/// Perform an assertion at compile time.
241///
242/// # Example
243///
244/// ```ignore
245/// use inkwell::support::const_assert;
246/// #[repr(transparent)]
247/// struct Foo {
248///     bar: &'static (),
249/// }
250/// const _: () = const_assert(size_of::<Foo>() == size_of::<&'static ()>());
251/// ```
252#[track_caller]
253#[inline(always)]
254pub(crate) const fn const_assert(assertion: bool) {
255    if !assertion {
256        panic!("Assertion Failed");
257    }
258}

260/// Asserts that a type has at least one niche. Having a niche means that `size_of::<T>() == size_of::<Option<T>>()`.
261///
262/// # Example
263///
264/// ```ignore
265/// use std::ptr::NonNull;
266/// use inkwell::support::assert_niche;
267///
268/// struct Foo {
269///     bar: NonNull<Foo>,
270/// }
271/// const _: () = assert_niche::<Foo>();
272/// ```
273#[track_caller]
274#[inline(always)]
275pub(crate) const fn assert_niche<T>() {
276    const_assert(size_of::<T>() == size_of::<Option<T>>() && align_of::<T>() == align_of::<Option<T>>());
277}
