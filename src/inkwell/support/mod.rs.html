18/// An owned LLVM String. Also known as a LLVM Message
19#[derive(Eq)]
20pub struct LLVMString {
21    pub(crate) ptr: *const c_char,
22}
23
24unsafe impl Send for LLVMString {}
25unsafe impl Sync for LLVMString {}
26
27impl LLVMString {
28    pub(crate) unsafe fn new(ptr: *const c_char) -> Self {
29        assert!(!ptr.is_null());
30        LLVMString { ptr }
31    }
32
33    /// This is a convenience method for creating a Rust `String`,
34    /// however; it *will* reallocate. `LLVMString` should be used
35    /// as much as possible to save memory since it is allocated by
36    /// LLVM. It's essentially a `CString` with a custom LLVM
37    /// deallocator
38    #[allow(clippy::inherent_to_string_shadow_display)]
39    pub fn to_string(&self) -> String {
40        (*self).to_string_lossy().into_owned()
41    }
42
43    /// This method will allocate a c string through LLVM
44    pub(crate) fn create_from_c_str(string: &CStr) -> LLVMString {
45        unsafe { LLVMString::new(LLVMCreateMessage(string.as_ptr() as *const _)) }
46    }
47
48    /// This method will allocate a c string through LLVM
49    pub(crate) fn create_from_str(string: &str) -> LLVMString {
50        debug_assert_eq!(string.as_bytes()[string.len() - 1], 0);
51
52        unsafe { LLVMString::new(LLVMCreateMessage(string.as_ptr() as *const _)) }
53    }
54}
55
56impl Deref for LLVMString {
57    type Target = CStr;
58
59    fn deref(&self) -> &Self::Target {
60        unsafe { CStr::from_ptr(self.ptr) }
61    }
62}

84        self.to_str()

94    fn drop(&mut self) {
95        unsafe {

101// Similar to Cow; however does not provide ability to clone
102// since memory is allocated by LLVM. Could use a better name
103// too. This is meant to be an internal wrapper only. Maybe
104// belongs in a private utils module.
105#[derive(Eq)]
106pub(crate) enum LLVMStringOrRaw {
107    Owned(LLVMString),
108    Borrowed(*const c_char),
109}
110
111impl LLVMStringOrRaw {
112    pub fn as_str(&self) -> &CStr {
113        match self {
114            LLVMStringOrRaw::Owned(llvm_string) => llvm_string.deref(),
115            LLVMStringOrRaw::Borrowed(ptr) => unsafe { CStr::from_ptr(*ptr) },
116        }
117    }
118}
119
120impl PartialEq for LLVMStringOrRaw {
121    fn eq(&self, other: &LLVMStringOrRaw) -> bool {
122        self.as_str() == other.as_str()
123    }
124}

134/// Returns the major, minor, and patch version of the LLVM in use
135#[llvm_versions(16..)]
136pub fn get_llvm_version() -> (u32, u32, u32) {
137    let mut major: u32 = 0;
138    let mut minor: u32 = 0;
139    let mut patch: u32 = 0;
140
141    unsafe { LLVMGetVersion(&mut major, &mut minor, &mut patch) };
142
143    (major, minor, patch)
144}

146/// Possible errors that can occur when loading a library
147#[derive(thiserror::Error, Debug, PartialEq, Eq, Clone, Copy)]
148pub enum LoadLibraryError {
149    /// The given path could not be converted to a [`&str`]
150    #[error("The given path could not be converted to a `&str`")]
151    UnicodeError,
152    /// The given path could not be loaded as a library
153    #[error("The given path could not be loaded as a library")]
154    LoadingError,
155}

157/// Permanently load the dynamic library at the given `path`.
158///
159/// It is safe to call this function multiple times for the same library.
160pub fn load_library_permanently(path: &Path) -> Result<(), LoadLibraryError> {
161    let filename = to_c_str(path.to_str().ok_or(LoadLibraryError::UnicodeError)?);
162
163    let error = unsafe { LLVMLoadLibraryPermanently(filename.as_ptr()) == 1 };
164    if error {
165        return Err(LoadLibraryError::LoadingError);
166    }
167
168    Ok(())
169}

184/// Search through all previously loaded dynamic libraries for `symbol`.
185///
186/// Returns an address of the symbol, if found
187pub fn search_for_address_of_symbol(symbol: &str) -> Option<usize> {
188    let symbol = to_c_str(symbol);
189
190    let address = unsafe { LLVMSearchForAddressOfSymbol(symbol.as_ptr()) };
191    if address.is_null() {
192        return None;
193    }
194    Some(address as usize)
195}

200    load_visible_symbols();

216/// This function takes in a Rust string and either:
217///
218/// A) Finds a terminating null byte in the Rust string and can reference it directly like a C string.
219///
220/// B) Finds no null byte and allocates a new C string based on the input Rust string.
221#[inline]
222pub(crate) fn to_c_str(mut s: &str) -> Cow<'_, CStr> {
223    if s.is_empty() {
224        s = "\0";
225    }
226
227    match CStr::from_bytes_until_nul(s.as_bytes()) {
228        Ok(c) => Cow::from(c),
229        // SAFETY: No internal 0 byte since already `FromBytesUntilNulError`
230        Err(_) => unsafe { Cow::from(CString::new(s.as_bytes()).unwrap_unchecked()) },
231    }
232}

234#[test]
235fn test_to_c_str() {
236    assert_eq!(to_c_str("my string"), Cow::<CStr>::Owned(c"my string".to_owned()));
237    assert_eq!(to_c_str("my\0string"), Cow::Borrowed(c"my"));
238    assert_eq!(to_c_str("my string\0"), Cow::Borrowed(c"my string"));
