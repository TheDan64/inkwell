5/// A trait for user-defined memory management in MCJIT.
6///
7/// Implementors can override how LLVM's MCJIT engine allocates memory for code
8/// and data sections. This is sometimes needed for:
9/// - custom allocators,
10/// - sandboxed or restricted environments,
11/// - capturing stack map sections (e.g., for garbage collection),
12/// - or other specialized JIT memory management requirements.
13///
14/// # StackMap and GC Integration
15///
16/// By examining the `section_name` argument in [`McjitMemoryManager::allocate_data_section`],
17/// you can detect sections such as `.llvm_stackmaps` (on ELF) or `__llvm_stackmaps`
18/// (on Mach-O). Recording the location of these sections may be useful for
19/// custom garbage collectors. For more information, refer to the [LLVM
20/// StackMaps documentation](https://llvm.org/docs/StackMaps.html#stack-map-section).
21///
22/// Typically, on Darwin (Mach-O), the stack map section name is `__llvm_stackmaps`,
23/// and on Linux (ELF), it is `.llvm_stackmaps`.
24pub trait McjitMemoryManager: std::fmt::Debug {
25    /// Allocates a block of memory for a code section.
26    ///
27    /// # Parameters
28    ///
29    /// * `size` - The size in bytes for the code section.
30    /// * `alignment` - The required alignment in bytes.
31    /// * `section_id` - A numeric ID that LLVM uses to identify this section.
32    /// * `section_name` - A name for this section, if provided by LLVM.
33    ///
34    /// # Returns
35    ///
36    /// Returns a pointer to the allocated memory. Implementors must ensure it is
37    /// at least `size` bytes long and meets `alignment` requirements.
38    fn allocate_code_section(
39        &mut self,
40        size: libc::uintptr_t,
41        alignment: libc::c_uint,
42        section_id: libc::c_uint,
43        section_name: &str,
44    ) -> *mut u8;
45
46    /// Allocates a block of memory for a data section.
47    ///
48    /// # Parameters
49    ///
50    /// * `size` - The size in bytes for the data section.
51    /// * `alignment` - The required alignment in bytes.
52    /// * `section_id` - A numeric ID that LLVM uses to identify this section.
53    /// * `section_name` - A name for this section, if provided by LLVM.
54    /// * `is_read_only` - Whether this data section should be read-only.
55    ///
56    /// # Returns
57    ///
58    /// Returns a pointer to the allocated memory. Implementors must ensure it is
59    /// at least `size` bytes long and meets `alignment` requirements.
60    fn allocate_data_section(
61        &mut self,
62        size: libc::uintptr_t,
63        alignment: libc::c_uint,
64        section_id: libc::c_uint,
65        section_name: &str,
66        is_read_only: bool,
67    ) -> *mut u8;
68
69    /// Finalizes memory permissions for all allocated sections.
70    ///
71    /// This is called once all sections have been allocated. Implementors can set
72    /// permissions such as making code sections executable or data sections
73    /// read-only.
74    ///
75    /// # Errors
76    ///
77    /// If any error occurs (for example, failing to set page permissions),
78    /// return an `Err(String)`. This error is reported back to LLVM as a C string.
79    fn finalize_memory(&mut self) -> Result<(), String>;
80
81    /// Cleans up or deallocates resources before the memory manager is destroyed.
82    ///
83    /// This is called when LLVM has finished using the memory manager. Any
84    /// additional allocations or references should be released here if needed.
85    fn destroy(&mut self);
86}

88/// Holds a boxed `McjitMemoryManager` and passes it to LLVM as an opaque pointer.
89///
90/// LLVM calls into the adapter using the extern "C" function pointers defined below.
91#[repr(transparent)]
92#[derive(Debug)]
93pub struct MemoryManagerAdapter {
94    pub memory_manager: Box<dyn McjitMemoryManager>,
95}
96const _: () = assert_niche::<MemoryManagerAdapter>();

100/// Adapter for `allocate_code_section`.
101///
102/// Called by LLVM with a raw pointer (`opaque`). Casts back to `MemoryManagerAdapter`
103/// and delegates to `allocate_code_section`.
104pub(crate) extern "C" fn allocate_code_section_adapter(
105    opaque: *mut libc::c_void,
106    size: libc::uintptr_t,
107    alignment: libc::c_uint,
108    section_id: libc::c_uint,
109    section_name: *const libc::c_char,
110) -> *mut u8 {
111    let adapter = unsafe { &mut *(opaque as *mut MemoryManagerAdapter) };
112    let sname = unsafe { c_str_to_str(section_name) };
113    adapter
114        .memory_manager
115        .allocate_code_section(size, alignment, section_id, sname)
116}

118/// Adapter for `allocate_data_section`.
119///
120/// Note that `LLVMBool` is `0` for false, and `1` for true. We check `!= 0` to
121/// interpret it as a bool.
122pub(crate) extern "C" fn allocate_data_section_adapter(
123    opaque: *mut libc::c_void,
124    size: libc::uintptr_t,
125    alignment: libc::c_uint,
126    section_id: libc::c_uint,
127    section_name: *const libc::c_char,
128    is_read_only: LLVMBool,
129) -> *mut u8 {
130    let adapter = unsafe { &mut *(opaque as *mut MemoryManagerAdapter) };
131    let sname = unsafe { c_str_to_str(section_name) };
132    adapter
133        .memory_manager
134        .allocate_data_section(size, alignment, section_id, sname, is_read_only != 0)
135}

137/// Adapter for `finalize_memory`.
138///
139/// If an error is returned, the message is converted into a C string and set in `err_msg_out`.
140pub(crate) extern "C" fn finalize_memory_adapter(
141    opaque: *mut libc::c_void,
142    err_msg_out: *mut *mut libc::c_char,
143) -> libc::c_int {
144    let adapter = unsafe { &mut *(opaque as *mut MemoryManagerAdapter) };
145    match adapter.memory_manager.finalize_memory() {
146        Ok(()) => 0,
147        Err(e) => {
148            let cstring = std::ffi::CString::new(e).unwrap_or_default();
149            unsafe {
150                *err_msg_out = cstring.into_raw();
151            }
152            1
153        },
154    }
155}

157/// Adapter for `destroy`.
158///
159/// Called when LLVM is done with the memory manager. Calls `destroy` and drops
160/// the adapter to free resources.
161pub(crate) extern "C" fn destroy_adapter(opaque: *mut libc::c_void) {
162    // Re-box to drop the adapter and its contents.
163    // SAFETY: `opaque` must have been allocated by Box<MemoryManagerAdapter>.
164    let mut adapter = unsafe { Box::from_raw(opaque as *mut MemoryManagerAdapter) };
165
166    // Clean up user-defined resources
167    adapter.memory_manager.destroy();
168
169    // Dropping `adapter` automatically frees the memory
170}

172/// Converts a raw C string pointer to a Rust `&str`.
173///
174/// # Safety
175///
176/// The caller must ensure `ptr` points to a valid, null-terminated UTF-8 string.
177/// If the string is invalid UTF-8 or `ptr` is null, an empty string is returned.
178unsafe fn c_str_to_str<'a>(ptr: *const libc::c_char) -> &'a str {
179    if ptr.is_null() {
180        ""
181    } else {
182        unsafe { std::ffi::CStr::from_ptr(ptr) }.to_str().unwrap_or("")
183    }
