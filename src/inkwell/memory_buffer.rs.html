18#[derive(Debug)]
19pub struct MemoryBuffer<'a> {
20    pub(crate) memory_buffer: LLVMMemoryBufferRef,
21    _phantom: PhantomData<&'a [u8]>,
22}
23
24// The backing memory is owned and managed by LLVM, such that it is valid throughout the buffer's
25// lifetime, hence 'static.
26impl MemoryBuffer<'static> {
27    /// Create a memory buffer from file content.
28    ///
29    /// Return `Err` with error message if failed.
30    pub fn create_from_file(path: &Path) -> Result<Self, LLVMString> {
31        let path = to_c_str(path.to_str().expect("Did not find a valid Unicode path string"));
32        let mut memory_buffer = ptr::null_mut();
33        let mut err_string = ptr::null_mut();
34
35        let is_err = unsafe {
36            LLVMCreateMemoryBufferWithContentsOfFile(path.as_ptr(), &mut memory_buffer, &mut err_string) == 1
37        };
38
39        if is_err {
40            unsafe {
41                return Err(LLVMString::new(err_string));
42            }
43        }
44
45        unsafe { Ok(Self::new(memory_buffer)) }
46    }
47
48    /// Create a memory buffer from stdin.
49    ///
50    /// Return `Err` with error message if failed.
51    pub fn create_from_stdin() -> Result<Self, LLVMString> {
52        let mut memory_buffer = ptr::null_mut();
53        let mut err_string = ptr::null_mut();
54
55        let is_err = unsafe { LLVMCreateMemoryBufferWithSTDIN(&mut memory_buffer, &mut err_string) == 1 };
56
57        if is_err {
58            unsafe {
59                return Err(LLVMString::new(err_string));
60            }
61        }
62
63        unsafe { Ok(Self::new(memory_buffer)) }
64    }
65
66    /// Create a memory buffer copied from a byte slice with a trailing nul byte.
67    ///
68    /// # Panics
69    ///
70    /// Panics if the input byte slice does not terminate with a nul byte.
71    pub fn create_from_memory_range_copy(input: &[u8], name: &str) -> Self {
72        assert_eq!(
73            input[input.len() - 1],
74            b'\0',
75            "input byte slice must terminate with a nul byte"
76        );
77
78        let name_c_string = to_c_str(name);
79
80        let memory_buffer = unsafe {
81            LLVMCreateMemoryBufferWithMemoryRangeCopy(
82                input.as_ptr() as *const libc::c_char,
83                // decremented since technically the nul byte is one past the end of the input.
84                input.len() - 1,
85                name_c_string.as_ptr(),
86            )
87        };
88
89        unsafe { Self::new(memory_buffer) }
90    }
91}
92
93impl<'a> MemoryBuffer<'a> {
94    pub unsafe fn new(memory_buffer: LLVMMemoryBufferRef) -> Self {
95        assert!(!memory_buffer.is_null());
96
97        Self {
98            memory_buffer,
99            _phantom: PhantomData,
100        }
101    }
102
103    pub fn as_mut_ptr(&self) -> LLVMMemoryBufferRef {
104        self.memory_buffer
105    }
106
107    /// Create a memory buffer from a byte slice with a trailing nul byte.
108    ///
109    /// # Panics
110    ///
111    /// Panics if the input byte slice does not terminate with a nul byte.
112    pub fn create_from_memory_range(input: &'a [u8], name: &str) -> Self {
113        assert_eq!(
114            input[input.len() - 1],
115            b'\0',
116            "input byte slice must terminate with a nul byte"
117        );
118
119        let name_c_string = to_c_str(name);
120
121        let memory_buffer = unsafe {
122            LLVMCreateMemoryBufferWithMemoryRange(
123                input.as_ptr() as *const libc::c_char,
124                // decremented since technically the nul byte is one past the end of the input.
125                input.len() - 1,
126                name_c_string.as_ptr(),
127                false as i32, // guaranteed to have nul-terminator by CStr
128            )
129        };
130
131        unsafe { MemoryBuffer::new(memory_buffer) }
132    }
133
134    /// Gets a byte slice of this [`MemoryBuffer`], containing the trailing nul byte.
135    pub fn as_slice(&self) -> &[u8] {
136        unsafe {
137            let start = LLVMGetBufferStart(self.memory_buffer);
138
139            // SAFETY: from LLVM `MemoryBuffer.h`:
140            // "this interface guarantees you can read one character past the end of the file,
141            // and that this character will read as '\0'."
142            //
143            // we include it here, so we can create a MemoryBuffer from the returned slice again.
144            slice::from_raw_parts(start as *const _, self.get_size())
145        }
146    }
147
148    /// Gets the byte size of this `MemoryBuffer`, counting the trailing nul byte.
149    pub fn get_size(&self) -> usize {
150        // buffer size does not include the trailing nul byte, hence incremented.
151        unsafe { LLVMGetBufferSize(self.memory_buffer) + 1 }
152    }
153
154    /// Convert this [`MemoryBuffer`] and optional [`Context`] into a [`BinaryFile`].
155    ///
156    /// The context is required if the resulting file is LLVM IR.
157    ///
158    /// Return `Err` with error message if failed.
159    pub fn create_binary_file(&self, context: Option<&Context>) -> Result<BinaryFile<'_>, LLVMString> {
160        let context = context.map_or(ptr::null_mut(), |c| c.raw());
161        let mut err_string = ptr::null_mut();
162
163        let binary_file = unsafe { LLVMCreateBinary(self.memory_buffer, context, &mut err_string) };
164
165        if binary_file.is_null() {
166            unsafe {
167                return Err(LLVMString::new(err_string));
168            }
169        }
170
171        unsafe { Ok(BinaryFile::new(binary_file)) }
172    }
173}
174
175impl<'a> Drop for MemoryBuffer<'a> {
176    fn drop(&mut self) {
177        unsafe {
178            LLVMDisposeMemoryBuffer(self.memory_buffer);
179        }
180    }
181}
